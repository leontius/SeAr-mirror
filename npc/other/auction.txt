function	script	F_AuctionWarper	{
	mes "[拍卖场 服务人员]";
	mes "请问你要去拍卖场吗?";
	next;
	if (select("是的:不是") == 1) {
		if (getarg(0) == 1 || getarg(0) == 4) {
			mes "[拍卖场 服务人员]"; 
			mes "太好了！那么，"; 
			mes "祝你拍卖愉快~"; 
		}else{
			mes "[拍卖场 服务人员]";
			mes "那么祝你一路顺风~";
		}
		close2;
		switch(getarg(0)) {
		case 1:
			warp "auction_01",179,53; end;
		case 2:
			warp "auction_01",21,43; end;
		case 3:
			warp "auction_02",151,23; end;
		case 4:
			warp "auction_02",43,24; end;
		}
	}
	mes "[拍卖场 服务人员]";
	mes "那么下一次请你再来呦";
	close;
}
